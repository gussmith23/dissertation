\begin{grammar}

<expr> ::= signal-const | signal-var  
\alt <unop> <expr> 
        \alt <binop> <expr> <expr>
        \alt <ternop> <expr> <expr> <expr>
        
<unop> ::=  `not' | `neg' | `reduce_and' | `reduce_xor'
    
<binop> ::= `and'  | `or' | `xor'
    \alt  `eq' | `ne' | `logic_and' | `logic_not' | `logic_or' 
    \alt  `mul' | `add' | `sub' 
    \alt  `dff'
    \alt `concat'

<ternop> ::=  `mux'  | `extract'

\end{grammar}
